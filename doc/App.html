before do
    @role = 'guest'
    admin = User.selectFromId(db, session[:user_id].to_i)
    if admin != nil 
        @admin = admin["admin"]
    else
        @admin = 0
    end
    @rut = request.fullpath
    if session[:user_id] != nil
        if @admin == admin["admin"]
            @role = 'admin'
        else
            @role = 'standard'
        end
    end
end
